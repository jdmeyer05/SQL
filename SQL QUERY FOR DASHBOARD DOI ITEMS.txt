DISTINCT
DO_UID,
DOI_RC_ACTIVE_STATUS AS ACTIVE_STATUS,
DOI_RC_EFFECTIVE_DATE AS EFFECTIVE_DATE,
DOI_RC_EXPIRATION_DATE AS EXPIRATION_DATE,
DOI_PAYOUT_STATUS AS PAYOUT_STATUS

FROM
EULAPPS.P2_DOI_REPORT_CENTERS_V


WHERE 
RC_USER_KEY = 
421201754
AND
DOI_TYPE = 'REV'

DO_UID
